79 void AButton::manageState()
80 {
81  sf::Vector2i pos = _mouse.getPosition(_window);
82  bool hovering = false;
83 
84  if (!_updateState) {
85  _state = NONE;
86  return;
87  }
88  if (pos.x >= _pos.x && pos.x < _pos.x + _size.x &&
89  pos.y >= _pos.y && pos.y < _pos.y + _size.y)
90  hovering = true;
91  if (_state != PRESSED) {
92  if (_mouse.isButtonPressed(_button) && _state == HOVERING) {
93  _state = PRESSED;
94  return;
95  }
96  } else if (!_mouse.isButtonPressed(_button)) {
97  _state = PRESSED;
98  if (hovering && _func)
99  _func();
100  } else
101  return;
102  if (hovering && !_mouse.isButtonPressed(_button)) {
103  _state = HOVERING;
104  return;
105  }
106  _state = NONE;
107 }
