212 std::pair<float, float> arc::Nibbler::findCoordSnakeBody(float x, float y)
213 {
214  const auto findObjLeft = std::find_if(_entities.begin(), _entities.end(), [x, y](std::shared_ptr<Entity> const &obj){

245  std::uniform_int_distribution<int> ranX {0, COLS_SNAKE - 1};

265 void arc::Nibbler::eatFruit()
266 {
267  std::default_random_engine re(std::chrono::system_clock::now().time_since_epoch().count());
268  std::uniform_int_distribution<int> ranX {0, COLS_SNAKE - 1};
269  std::uniform_int_distribution<int> ranY {0, ROWS_SNAKE - 1};
270  float newX = 0;
271  float newY = 0;
272  for (auto &fruit : _fruits) {
273  if (fruit->x == _snakeHead->x && fruit->y == _snakeHead->y) {
274  _score++;
275  this->addSnakeBody();
276  if (_gameOver)
277  return;
278  do {
279  newX = ranX(re);
280  newY = ranY(re);
281  } while (!invalidCoordonate(newX, newY));
282  fruit->x = newX;
283  fruit->y = newY;
284  }
285  }
286 }
