126 void InputZone::update(const sf::Event &event)
127 {
128  if (!_hasFocus)
129  return;
130  if (event.type == sf::Event::TextEntered) {
131  if (event.text.unicode == '\b') {
132  removeChar();
133  } else if (event.text.unicode == 127) {
134  _cursorPos++;
135  removeChar();
136  } else if (event.text.unicode == '\r' || event.text.unicode == '\t') {
137  _hasFocus = false;

139  addChar(event.text.unicode);

141  } else if (event.type == sf::Event::KeyPressed) {
142  if (event.key.code == sf::Keyboard::Left && _cursorPos > 0) {
143  _cursorPos--;

145  } else if (event.key.code == sf::Keyboard::Right) {
146  _cursorPos++;
147  _clock.restart();
148  }
