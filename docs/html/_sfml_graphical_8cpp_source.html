140 void SfmlGraphical::checkEvents()
141 {
142  static std::pair<Event::Type, Event::Key> event;
143  _eventType = Event::NO_EVENT;
144  _keyPressed = Event::NONE;
145 
146  if (!_window.isOpen()) {
147  _eventType = Event::QUIT;
148  return;
149  }
150  while (_window.pollEvent(_event)) {
151  if (_event.type == sf::Event::Closed) {
152  _window.close();
153  _eventType = Event::QUIT;
154  } else if (_event.type == sf::Event::KeyPressed ||
155  _event.type == sf::Event::KeyReleased) {
156  if (_event.type == sf::Event::KeyPressed)
157  _eventType = Event::KEY_PRESSED;
158  else
159  _eventType = Event::KEY_RELEASED;
160  if (_equivalentKeys.count(_event.key.code))
161  _keyPressed = _equivalentKeys.at(_event.key.code);
162  else
163  _keyPressed = Event::Key::NONE;
164  } else if (_event.type == sf::Event::MouseButtonPressed) {
165  _eventType = Event::MOUSE_PRESSED;
166  switch (_event.mouseButton.button) {
167  case (sf::Mouse::Button::Left):
168  _keyPressed = Event::MOUSE_1;
169  break;
170  case (sf::Mouse::Button::Middle):
171  _keyPressed = Event::MOUSE_2;
172  break;
173  case (sf::Mouse::Button::Right):
174  _keyPressed = Event::MOUSE_3;
175  break;
176  default:
177  _keyPressed = Event::NONE;
178  break;
179  }
180  }
181 
182  if (_scenes.count(_scene))
183  _scenes.at(_scene)->update(_event);
184 
185  event.first = _eventType;
186  event.second = _keyPressed;
187  }
188 }

290 void SfmlGraphical::setSpriteSize(sf::Sprite &sprite, const sf::Vector2f &size)
291 {
292  sf::FloatRect rect = sprite.getLocalBounds();
293  sf::Vector2f scale;
294 
295  if (!rect.width || !rect.height || (rect.height == size.x && rect.width == size.y))
296  return;
297  scale.x = size.x / rect.width;
298  scale.y = size.y / rect.height;
