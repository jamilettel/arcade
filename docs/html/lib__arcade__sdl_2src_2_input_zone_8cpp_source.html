130 void InputZone::update(const SDL_Event &event)
131 {
132  if (event.type == SDL_MOUSEBUTTONDOWN && event.button.button == SDL_BUTTON_LEFT)
133  _mouseButtonPressed = true;
134  else if (event.type == SDL_MOUSEBUTTONUP && event.button.button == SDL_BUTTON_LEFT)
135  _mouseButtonPressed = false;
136  if (!_hasFocus)
137  return;
138  if (event.type == SDL_TEXTINPUT) {
139  if (event.text.text[0] >= ' ' && event.text.text[0] <= '~')
140  addChar(event.text.text[0]);
141  } else if (event.type == SDL_KEYDOWN) {
142  if (event.key.keysym.sym == SDLK_DELETE) {
143  _cursorPos++;
144  removeChar();
145  } else if (event.key.keysym.sym == SDLK_BACKSPACE) {
146  removeChar();
147  } else if (event.key.keysym.sym == SDLK_RETURN || event.key.keysym.sym == SDLK_TAB) {
148  _hasFocus = false;
149  } else if (event.key.keysym.sym == SDLK_LEFT && _cursorPos > 0) {
150  _cursorPos--;
151  _currentTicks = SDL_GetTicks();
152  } else if (event.key.keysym.sym == SDLK_RIGHT) {
153  _cursorPos++;
154  _currentTicks = SDL_GetTicks();
155  }
156  if (_cursorPos > _content.size()) {
157  _cursorPos = _content.size();
158  }
159  }
160 }

167 
168  if (_mouseButtonPressed) {

170  mousePos.y >= globalBounds.y && mousePos.y < globalBounds.y + globalBounds.h)
171  _hasFocus = true;
172  else

174  }
175  if (_hasFocus && _currentOutlineColor != _focusOutline) {
176  _currentOutlineColor = _focusOutline;

178  } else if (!_hasFocus && _currentOutlineColor != _unfocusedOutline) {
179  _currentOutlineColor = _unfocusedOutline;
180  _colorOutline.setColor(_currentOutlineColor);
181  }
182  _rs.setColor(_colorBackground.getColor());
183  _rs.setOutlineColor(_colorOutline.getColor());
