44 void NcursesMainMenu::update()
45 {
46  std::pair<Event::Type, Event::Key> event;
47 
48  event.first = _lib.getEventType();
49  event.second = _lib.getKeyPressed();
50  if (event.second == Event::Key::DOWN) {
51  if (_activeMenu == 1 && _chosenGame + 1 != _listGames->size()) {
52  _chosenGame++;
53  _ftGames.value()(_listGames->at(_chosenGame));
54  }
55  else if (_activeMenu == 2 && _chosenGraphics + 1 != _listGraphics->size())
56  _chosenGraphics++;
57  }
58  if (event.second == Event::Key::UP) {
59  if (_activeMenu == 1 && _chosenGame != 0) {
60  _chosenGame--;
61  _ftGames.value()(_listGames->at(_chosenGame));
62  }
63  else if (_activeMenu == 2 && _chosenGraphics != 0)
64  _chosenGraphics--;
65  }
66  if (event.second == Event::Key::LEFT)
67  _activeMenu = 1;
68  if (event.second == Event::Key::RIGHT)
69  _activeMenu = 2;
70  if (event.second == Event::Key::ENTER && _activeMenu == 1) {
71  _ftGames.value()(_listGames->at(_chosenGame));
72  _lib._playFct.value()();
73  }
74  if (event.second == Event::Key::ENTER && _activeMenu == 2) {
75  _ftGraphics.value()(_listGraphics->at(_chosenGraphics));
76  }
77  if (event.second == Event::Key::U) {
78  this->promptUsername();
79  }
80 }
