117 void NcursesGraphical::checkEvents()
118 {
119  int key_pressed;

240 void NcursesGraphical::addColor(Color color)
241 {
242  static short id = 0;
243  if (_colorIndex.find(color) != _colorIndex.end())
244  return;
245  if (id > COLORS - 1)
246  id = 0;
247  short rFinal = color.r * 4;
248  short gFinal = color.g * 4;
249  short bFinal = color.b * 4;
250  if (rFinal > 1000) rFinal = 1000;
251  if (gFinal > 1000) gFinal = 1000;
252  if (bFinal > 1000) bFinal = 1000;
253  init_color(id, rFinal, gFinal, bFinal);
254  _colorIndex[color] = id;
255  id++;
256 }
