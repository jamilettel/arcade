267 void NcursesGraphical::addColor(Color color)
268 {
269  static short id = 0;
270  if (_colorIndex.find(color) != _colorIndex.end())
271  return;
272  if (id > COLORS - 1)
273  id = 0;
274  short rFinal = color.r * 4;
275  short gFinal = color.g * 4;
276  short bFinal = color.b * 4;
277  if (rFinal > 1000) rFinal = 1000;
278  if (gFinal > 1000) gFinal = 1000;
279  if (bFinal > 1000) bFinal = 1000;
280  init_color(id, rFinal, gFinal, bFinal);
281  _colorIndex[color] = id;
282  id++;
283 }
