137 void SDLGraphical::checkEvents()
138 {
139  static std::pair<Event::Type, Event::Key> event;
140  _eventType = Event::NO_EVENT;
141  _keyPressed = Event::NONE;
142 
143  if (!_window.isOpen()) {
144  _eventType = Event::QUIT;
145  return;
146  }
147  while (SDL_PollEvent(&_event)) {
148  if (_event.type == SDL_QUIT) {
149  _window.close();
150  _eventType = Event::QUIT;
151  } else if (_event.type == SDL_KEYDOWN ||
152  _event.type == SDL_KEYUP) {
153  if (_event.type == SDL_KEYDOWN)
154  _eventType = Event::KEY_PRESSED;
155  else
156  _eventType = Event::KEY_RELEASED;
157  if (_equivalentKeys.count(_event.key.keysym.sym))
158  _keyPressed = _equivalentKeys.at(_event.key.keysym.sym);
159  else
160  _keyPressed = Event::Key::NONE;
161  } else if (_event.type == SDL_MOUSEBUTTONDOWN) {
162  _eventType = Event::MOUSE_PRESSED;
163  switch (_event.button.button) {
164  case (SDL_BUTTON_LEFT):
165  _keyPressed = Event::MOUSE_1;
166  break;
167  case (SDL_BUTTON_MIDDLE):
168  _keyPressed = Event::MOUSE_2;
169  break;
170  case (SDL_BUTTON_RIGHT):
171  _keyPressed = Event::MOUSE_3;
172  break;
173  default:
174  _keyPressed = Event::NONE;
175  break;
176  }
177  }
178 
179  if (_scenes.count(_scene))
180  _scenes.at(_scene)->update(_event);
181 
182  event.first = _eventType;
183  event.second = _keyPressed;
184  }
185 }
