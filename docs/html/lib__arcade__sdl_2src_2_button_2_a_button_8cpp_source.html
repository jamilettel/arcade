77 void AButton::manageState()
78 {
79  MySDL::Vector pos;
80  SDL_GetMouseState(&pos.x, &pos.y);
81  bool hovering = false;
82 
83  if (!_updateState) {
84  _state = NONE;
85  return;
86  }
87  if (pos.x >= _pos.x && pos.x < _pos.x + _size.x &&
88  pos.y >= _pos.y && pos.y < _pos.y + _size.y)
89  hovering = true;
90  if (_state != PRESSED) {
91  if (_buttonPressed && _state == HOVERING) {
92  _state = PRESSED;
93  return;
94  }
95  } else if (!_buttonPressed) {
96  _state = PRESSED;
97  if (hovering && _func)
98  _func();
99  } else
100  return;
101  if (hovering && !_buttonPressed) {
102  _state = HOVERING;
103  return;
104  }
105  _state = NONE;
