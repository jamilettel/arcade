326 {

335  std::istringstream lineParse(line);
336  std::pair<std::string, std::string> score;
337  lineParse >> score.first;

339  if (score.first != "" && score.second != "")
340  _bestScoresGame.emplace_back(score);
341  }
342  std::sort(_bestScoresGame.begin(), _bestScoresGame.end(), [](const auto &a, const auto &b){
343  return std::stoi(a.second.c_str()) >= std::stoi(b.second.c_str());
344  });
345  if (_bestScoresGame.size() > 10)
346  _bestScoresGame.erase(_bestScoresGame.begin() + 10, _bestScoresGame.end());
347  fileScores.close();
